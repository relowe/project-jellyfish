definitions
    # Structure definitions
    structure Person
        name: text
        age: number
    end structure

    # Global variable definitions
    adult_age: number = 18

    # Function definitions
    function greet(p: person) returns nothing
        print("Hello ", p.name, "!")
    end function
end definitions

program
    person1: person = {"Bob", 23}
    greet(person1)

    if person1.age >= adult_age then
        print("You are an adult")
    else
        print("You are not an adult yet")
    end if
end program
