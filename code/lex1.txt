definitions
    structure node
        value: number
        next: link to node
    end structure

    function collatz(n: number) returns number
        if n mod 2 = 0 then
            return n / 2
        else
            return 3 * n + 1
        end if
    end function

    function max(arr: array [10] of number) returns number
        max: number = arr[1]
        index: number = 2

        while index <= 10
            print("At index ", index, " with value ", arr[index])

            if arr[index] > max then
                max = arr[index]
            end if
            index = index + 1
        end while

        return max
    end function

end definitions
