definitions
    function bubble_sort(arr: changeable array [1 to 10] of number) returns nothing
        swapped : number = 1
        temp: number = 0
        while swapped = 1
            swapped = 0
            i : number = 1
            repeat 9 times
                if arr[i] > arr[i+1] then
                    temp = arr[i]
                    arr[i] = arr[i+1]
                    arr[i+1] = temp
                    swapped = 1
                end if
                i = i + 1
            end repeat
        end while
    end function
end definitions

program
    nums : array [10] of number = [7, 2, 1, 8, 6, 3, 5, 4, 9, 10]
    display("Before sorting:", nums)
    bubble_sort(nums)
    display("After sorting:", nums)
end program
